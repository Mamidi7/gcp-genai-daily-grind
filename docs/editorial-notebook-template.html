Key functions explained block by block

1. Configuration

import os from google.cloud import bigquery from dotenv import load_dotenv load_dotenv() PROJECT = os.getenv("GOOGLE_CLOUD_PROJECT") DATASET = os.getenv("BQ_DATASET") if not PROJECT or not DATASET: raise ValueError("GOOGLE_CLOUD_PROJECT and BQ_DATASET required")

2. Core Export Logic

def export_partition(table_id, partition_dt): client = bigquery.Client() sql = f""" SELECT * FROM `{PROJECT}.{DATASET}.{table_id}` WHERE _PARTITIONDATE = @partition_dt """ job_config = bigquery.QueryJobConfig( query_parameters=[ bigquery.ScalarQueryParameter( "partition_dt", "DATE", partition_dt ) ] ) job = client.query(sql, job_config=job_config) df = job.result().to_dataframe() return df
Always use @partition_dt parameterized — prevents SQL injection and enables cache reuse.
