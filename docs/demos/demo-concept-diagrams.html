# main.py — Flask app with health endpoint from flask import Flask app = Flask(__name__) @app.route("/") def hello(): return {"message": "Hello from Cloud Run!"} @app.route("/health") def health(): return {"status": "healthy"} if __name__ == "__main__": app.run(host="0.0.0.0", port=8080)
# requirements.txt — only two dependencies flask==3.1.0 gunicorn==23.0.0
